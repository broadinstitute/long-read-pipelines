  • same_barcode_per_read (Boolean, default=false)
  • shard_width (Int, default=25)

  • t_10_AdjustUmiSequenceWithAdapterAlignment.existing_umi_tag (String, default="ZU")
  • t_10_AdjustUmiSequenceWithAdapterAlignment.new_umi_tag (String, default="JX")
